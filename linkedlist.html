public class Node{
    public int data;
    public Node next;

    Node(int data) {
        this.data = data;
        this.next = null;
    }
}

public class LinkedList{
    public Node start = null;

    public LinkedList() {}

    public boolean isEmpty() {
        return start == null;
    }

    public void add(int data) {
        Node node = new Node(data);
        if (start == null) {
            start = node;
            return;
        }
        Node curr = start;
        while (curr.next != null) curr = curr.next;
        curr.next = node;
    }

    public void display() {
        Node node = start;
        System.out.println("START -> ");
        while (node != null) {
            System.out.print(node.data + " -> ");
            node = node.next;
        }
        System.out.println("END");
    }

    public int length() {
        int count = 0;
        Node node = start;
        while (node != null) {
            count++;
            node = node.next;
        }
        return count;
    }

    public void delete(int index) {
        Node prev = null;
        Node curr = start;
        index--;
        while (curr != null && index >= 0) {
            prev = curr;
            curr = curr.next;
            index--;
        }
        if (curr != null && prev != null) {
            prev.next = curr.next;
        }
    }

    public void reverse() {
        Node first = null;
        Node middle = start;
        Node last = start; // trick to avoid if last == null condition on every iteration
        while (middle != null) {
            last = last.next; // trick to avoid if last == null condition on every iteration
            middle.next = first;
            first = middle;
            middle = last;
        }
        start = first;
    }
}
