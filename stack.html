public class Element {
  public int data;
  public Element next;

  public Element(int data) {
    this.data = data;
    this.next = null;
  }
}

public class Stack{
  private Element top;

  public Stack() {}

  public void push(int data) {
    Element e = new Element(data);
    e.next = top;
    top = e;
  }

  public int pop() {
    if (top == null) {
      System.out.println("Stack is empty");
      return -9999;
    }
    int data = top.data;
    top = top.next;
    return data;
  }

  public boolean isEmpty() {
    return top == null;
  }

  public int size() {
    int count = 0;
    Element p = top;
    while (p != null) {
      count++;
      p = p.next;
    }
    return count;
  }

  public void display() {
    Element p = top;
    System.out.print("TOP");
    while (p != null) {
      System.out.print(" -> " + p.data);
      p = p.next;
    }
    System.out.println();
  }
}
