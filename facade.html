Facade Design Pattern

  • Structural Design Pattern
  • This pattern should be used when you come across code which is open for duplication, i.e., the same responsibility is being added in multiple places in our application.
  • Decouple business logic and implementation details of an application or third party classes which are hard to comprehend and maintain.
  • Centralize all responsibilities and restrict direct access to inner workings of the library.
Code 
// Facade Design Pattern
public class Main {
  public static void main(String[] args) {
    Facade1 f1 = new Facade1();
    Facade2 f2 = new Facade2();
    f1.foo312();
    f2.bar231();
  }
}


public class Subsystem1 {
  public Subsystem1() {}
  public void task1() {
    System.out.println("Working on process 1.1");
  }
  public void task2() {
    System.out.println("Working on process 1.2");
  }
  public void task3() {
    System.out.println("Working on process 1.3");
  }
}


public class Subsystem2 {
  public Subsystem2() {}
  public void task1() {
    System.out.println("Working on process 2.1");
  }
  public void task2() {
    System.out.println("Working on process 2.2");
  }
  public void task3() {
    System.out.println("Working on process 2.3");
  }
}


public class Subsystem3 {
  public Subsystem3() {}
  public void task1() {
    System.out.println("Working on process 3.1");
  }
  public void task2() {
    System.out.println("Working on process 3.2");
  }
  public void task3() {
    System.out.println("Working on process 3.3");
  }
}

public class Facade1 {
  private Subsystem1 c1 = new Subsystem1(); 
  private Subsystem2 c2 = new Subsystem2(); 
  private Subsystem3 c3 = new Subsystem3(); 

  public Facade1() {}

  public void foo312() {
    c1.task3();
    c2.task1();
    c3.task2();
  }
}

public class Facade2 {
  private Subsystem1 c1 = new Subsystem1(); 
  private Subsystem2 c2 = new Subsystem2(); 
  private Subsystem3 c3 = new Subsystem3(); 

  public Facade2() {}

  public void bar231() {
    c1.task2();
    c2.task3();
    c3.task1();
  }
}
